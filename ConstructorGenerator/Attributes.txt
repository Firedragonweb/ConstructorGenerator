using System;

namespace ConstructorGenerator
{
     [AttributeUsage(AttributeTargets.Field | AttributeTargets.Property)]
     internal class ConstructorDependencyAttribute : Attribute
     {
          internal ConstructorDependencyAttribute(bool isOptional = false)
          {
                IsOptional = isOptional;
          }
     
          public bool IsOptional { get; set; }
     }
     
     [AttributeUsage(AttributeTargets.Class)]
     internal class GenerateBaseConstructorCallAttribute : Attribute
     {
     
     }
     
     internal class GeneratedConstructorSettings
     {
          public ConstructorAccessibility Accessibility { get; set; } = ConstructorAccessibility.Public;
     }
     
     internal enum ConstructorAccessibility
     {
           Private,
           Protected,
           Internal,
           Public
     }
}
